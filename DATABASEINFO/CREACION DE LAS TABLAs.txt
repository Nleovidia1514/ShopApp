create table provider(
idprovider INTEGER NOT NULL PRIMARY KEY,
pname Varchar(20),
padress varchar (50));

create table client(
idclient Varchar(20) PRIMARY KEY,
cname Varchar(30),
cadress Varchar(100),
cmoney Numeric(10,2));

create table product(
idproduct INTEGER NOT NULL PRIMARY KEY,
pdescription Varchar(100),
buyprice Numeric(10,2),
sellprice Numeric(10,2),
restock INTEGER,
stock  INTEGER,
idprovider INTEGER NOT NULL,
FOREIGN KEY (idprovider) references provider(idprovider)
ON DELETE NO ACTION ON UPDATE CASCADE);

create table ocupation(
idocupation INTEGER NOT NULL PRIMARY KEY,
oname Varchar(40),
osalary Numeric(10,2));

create table employee(
idemployee INTEGER NOT NULL PRIMARY KEY,
ename Varchar(40),
eadress Varchar(60),
idocupation INTEGER NOT NULL,
FOREIGN KEY(idocupation) references ocupation(idocupation) 
ON DELETE NO ACTION ON UPDATE CASCADE,
sex Varchar(1));

create table bill(
idbill INTEGER NOT NULL PRIMARY KEY,
bdate timestamp,
idclient Varchar(20), 
FOREIGN KEY(idclient) references client(idclient) 
ON DELETE NO ACTION ON UPDATE CASCADE,
idemployee INTEGER NOT NULL, 
FOREIGN KEY(idemployee) references employee(idemployee) 
ON DELETE NO ACTION ON UPDATE CASCADE,
subtotal Numeric(12,2),
total Numeric(12,2));

create table bill_product(
idbill INTEGER NOT NULL, FOREIGN KEY(idbill) references bill(idbill) 
ON DELETE NO ACTION ON UPDATE CASCADE,
idproduct INTEGER NOT NULL, FOREIGN KEY(idproduct) references product(idproduct) 
ON DELETE NO ACTION ON UPDATE CASCADE,
quantity INTEGER,
total Numeric(13,2));
